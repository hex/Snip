# Snip v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a working, personal-grade macOS menu-bar app that inserts text snippets into the frontmost app via a CleanShot-flavored radial menu triggered by holding the middle mouse button (or a keyboard hotkey).

**Architecture:** A `SnipKit` Swift package holds all pure, Foundation-only logic (models, radial geometry, token resolution, JSON persistence, coordinate math) and is unit-tested with `swift test`. A thin `Snip` Xcode app target holds every AppKit/SwiftUI/CGEventTap surface and depends on `SnipKit`. The transient overlay is a borderless, non-activating `NSPanel` that is **display-only** — all selection input is driven by a global `CGEventTap`, never by SwiftUI gestures (forced by event consumption).

**Tech Stack:** Swift 5.9, AppKit (`NSStatusItem`, `NSPanel`, `NSVisualEffectView`, `CGEventTap`, `CGEvent`, `NSPasteboard`), SwiftUI (overlay content + windows), XcodeGen (`project.yml` → `.xcodeproj`), Swift Package Manager (`SnipKit`), XCTest.

## Global Constraints

Every task's requirements implicitly include this section. Values are copied verbatim from `docs/superpowers/specs/2026-07-09-snip-design.md`.

- **Minimum OS:** macOS 14.0 (Sonoma). Driven by `@Observable`.
- **App name:** `Snip`. **Bundle id:** `ai.symbiotica.Snip`.
- **Not sandboxed.** No Mac App Store. `com.apple.security.app-sandbox = false`.
- **Signing:** dev builds MUST be signed with a **stable** Developer ID / team and run from a **stable path** (TCC identifies the app by signature + path; churny DerivedData identities silently revoke Accessibility). Set `DEVELOPMENT_TEAM` in `project.yml` to your Apple Developer Team ID.
- **`SnipKit` imports Foundation only** — never AppKit/SwiftUI. (Invariant: keeps it pure + fast to test.)
- **Overlay is display-only** — the panel sets `ignoresMouseEvents = true` and all radial selection comes from `EventTapEngine`. Never route overlay selection through SwiftUI gestures. (Invariant: consumed tap events never reach any window.)
- **Insertion is paste-and-restore** with a `changeCount`-guarded restore and `org.nspasteboard` transient markers.
- **Caret offsets are measured in grapheme clusters** (Swift `String.count`), never UTF-16.

```swift
// ABOUTME: Loads and atomically saves the SnippetLibrary as JSON on disk.
// ABOUTME: Returns an empty library when absent and migrates older schema versions forward.
import Foundation

public final class SnippetStore {
    private let fileURL: URL

    public init(fileURL: URL) {
        self.fileURL = fileURL
    }

    public func load() throws -> SnippetLibrary {
        guard FileManager.default.fileExists(atPath: fileURL.path) else { return .empty }
        let data = try Data(contentsOf: fileURL)
        var library = try JSONDecoder().decode(SnippetLibrary.self, from: data)
        library = migrate(library)
        return library
    }

    public func save(_ library: SnippetLibrary) throws {
        try FileManager.default.createDirectory(
            at: fileURL.deletingLastPathComponent(), withIntermediateDirectories: true)
        let encoder = JSONEncoder()
        encoder.outputFormatting = [.prettyPrinted, .sortedKeys]
        let data = try encoder.encode(library)
        try data.write(to: fileURL, options: .atomic)
    }

    private func migrate(_ library: SnippetLibrary) -> SnippetLibrary {
        // No historical versions yet; bump schemaVersion so re-saves record current.
        var migrated = library
        migrated.schemaVersion = SnippetLibrary.currentSchemaVersion
        return migrated
    }
}
```

```swift
// ABOUTME: Resolves a snippet template into final text at fire time.
// ABOUTME: Expands {date}/{time}/{clipboard} and extracts the $| caret marker (grapheme-aware).
import Foundation

public struct ResolvedSnippet: Equatable {
    public let text: String
    public let caretOffsetFromEnd: Int
    public init(text: String, caretOffsetFromEnd: Int) {
        self.text = text
        self.caretOffsetFromEnd = caretOffsetFromEnd
    }
}

public struct TokenResolver {
    private let now: () -> Date
    private let locale: Locale
    private let timeZone: TimeZone
    private let clipboard: () -> String?

    public init(now: @escaping () -> Date = Date.init,
                locale: Locale = .current,
                timeZone: TimeZone = .current,
                clipboard: @escaping () -> String? = { nil }) {
        self.now = now
        self.locale = locale
        self.timeZone = timeZone
        self.clipboard = clipboard
    }

    public func resolve(_ body: String) -> ResolvedSnippet {
        // Split at the author's first $| so caret intent survives token expansion.
        let parts = body.components(separatedBy: "$|")
        let hasMarker = parts.count > 1
        let prefix = parts[0]
        let suffix = hasMarker ? parts.dropFirst().joined(separator: "$|") : ""

        let resolvedPrefix = expandTokens(prefix)
        let resolvedSuffix = expandTokens(suffix)
        let text = resolvedPrefix + resolvedSuffix
        let caretOffsetFromEnd = hasMarker ? resolvedSuffix.count : 0
        return ResolvedSnippet(text: text, caretOffsetFromEnd: caretOffsetFromEnd)
    }

    private func expandTokens(_ s: String) -> String {
        s.replacingOccurrences(of: "{date}", with: formatted(dateStyle: .medium, timeStyle: .none))
         .replacingOccurrences(of: "{time}", with: formatted(dateStyle: .none, timeStyle: .short))
         .replacingOccurrences(of: "{clipboard}", with: clipboard() ?? "")
    }

    private func formatted(dateStyle: DateFormatter.Style, timeStyle: DateFormatter.Style) -> String {
        let f = DateFormatter()
        f.locale = locale
        f.timeZone = timeZone
        f.dateStyle = dateStyle
        f.timeStyle = timeStyle
        return f.string(from: now())
    }
}
```

## Task 6: RadialSession — pointer → wedge selection (SnipKit, TDD)

**Files:**

- Test: `SnipKit/Tests/SnipKitTests/RadialSessionTests.swift`

**Interfaces:**
- Produces:
  - `enum RadialSelection: Equatable { case none; case wedge(Int) }`
  - `struct RadialSession { init(wedgeCount: Int, deadZoneRadius: Double, hysteresisDegrees: Double); func selection(dx: Double, dy: Double, previous: RadialSelection) -> RadialSelection }`
  - Convention: `dx` is rightward, `dy` is **downward** (screen coords). Wedge 0 is centered straight up (12 o'clock); indices increase clockwise. Inside `deadZoneRadius` ⇒ `.none`.

```swift
    }

    func testStraightRightIsWedgeTwo() {   // 90° clockwise from up, 8 wedges → index 2
```

```swift
// ABOUTME: Pure geometry mapping a pointer vector (from the bloom anchor) to a wedge selection.
// ABOUTME: Wedge 0 points up; indices increase clockwise; a center dead zone yields .none.
import Foundation

public enum RadialSelection: Equatable {
    case none
    case wedge(Int)
}

public struct RadialSession {
    private let wedgeCount: Int
    private let deadZoneRadius: Double
    private let hysteresisRadians: Double

    public init(wedgeCount: Int, deadZoneRadius: Double, hysteresisDegrees: Double) {
        self.wedgeCount = wedgeCount
        self.deadZoneRadius = deadZoneRadius
        self.hysteresisRadians = hysteresisDegrees * .pi / 180
    }

    public func selection(dx: Double, dy: Double, previous: RadialSelection) -> RadialSelection {
        if (dx * dx + dy * dy).squareRoot() < deadZoneRadius { return .none }

        let wedgeAngle = 2 * Double.pi / Double(wedgeCount)
        // Angle clockwise from straight up. dy is downward, so up is -dy.
        var angle = atan2(dx, -dy)
        if angle < 0 { angle += 2 * .pi }

        let raw = Int((angle / wedgeAngle).rounded()) % wedgeCount

        // Hysteresis: keep the previous wedge unless we've moved past its boundary + margin.
        if case let .wedge(prev) = previous, prev != raw {
            let prevCenter = Double(prev) * wedgeAngle
            var delta = abs(angle - prevCenter)
            if delta > .pi { delta = 2 * .pi - delta }
            if delta < wedgeAngle / 2 + hysteresisRadians { return .wedge(prev) }
        }
        return .wedge(raw)
    }
}
```

```swift
// ABOUTME: Also clamps the ring's origin so the whole overlay stays on-screen (notch/menu-bar safe).
import CoreGraphics

public enum ScreenGeometry {
    public static func cocoaPoint(fromQuartz p: CGPoint, primaryScreenHeight: CGFloat) -> CGPoint {
        CGPoint(x: p.x, y: primaryScreenHeight - p.y)
    }

        var x = center.x - size / 2

        y = min(max(y, visibleFrame.minY), visibleFrame.maxY - size)
        return CGPoint(x: x, y: y)
    }
```

```swift
// ABOUTME: Owns the global CGEventTaps that detect the trigger and stream pointer motion.
// ABOUTME: Consumes middle-button events; drives bloom/pointer/commit/cancel callbacks; runs off-main.
import AppKit
import SnipKit

final class EventTapEngine {
    static let magicUserData: Int64 = 0x5314_9    // stamps our synthetic events so we ignore them

    private let config: TriggerConfig
    private let permissions: PermissionsCoordinator
    private let onBloom: (CGPoint) -> Void
    private let onPointer: (RadialSelection) -> Void
    private let onCommit: (RadialSelection) -> Void
    private let onCancel: () -> Void

    private var tap: CFMachPort?
    private var runLoopSource: CFRunLoopSource?
    private var thread: Thread?

    private let session = RadialSession(wedgeCount: 8, deadZoneRadius: 24, hysteresisDegrees: 6)
    private var isOpen = false
    private var anchor = CGPoint.zero
    private var selection: RadialSelection = .none

    init(config: TriggerConfig, permissions: PermissionsCoordinator,
         onBloom: @escaping (CGPoint) -> Void, onPointer: @escaping (RadialSelection) -> Void,
         onCommit: @escaping (RadialSelection) -> Void, onCancel: @escaping () -> Void) {
        self.config = config; self.permissions = permissions
        self.onBloom = onBloom; self.onPointer = onPointer
        self.onCommit = onCommit; self.onCancel = onCancel
    }

    func start() -> Bool {
        guard permissions.isTrusted else { return false }
        let mask: CGEventMask =
            (1 << CGEventType.otherMouseDown.rawValue) |
            (1 << CGEventType.otherMouseUp.rawValue) |
            (1 << CGEventType.otherMouseDragged.rawValue) |
            (1 << CGEventType.mouseMoved.rawValue)
        let callback: CGEventTapCallBack = { _, type, event, refcon in
            let engine = Unmanaged<EventTapEngine>.fromOpaque(refcon!).takeUnretainedValue()
            return engine.handle(type: type, event: event)
        }
        guard let tap = CGEvent.tapCreate(
            tap: .cgSessionEventTap, place: .headInsertEventTap, options: .defaultTap,
            eventsOfInterest: mask, callback: callback,
            userInfo: Unmanaged.passUnretained(self).toOpaque()) else { return false }
        self.tap = tap
        let src = CFMachPortCreateRunLoopSource(nil, tap, 0)
        runLoopSource = src
        let t = Thread { [weak self] in
            guard let self, let src = self.runLoopSource, let tap = self.tap else { return }
            CFRunLoopAddSource(CFRunLoopGetCurrent(), src, .commonModes)
            CGEvent.tapEnable(tap: tap, enable: true)
            CFRunLoopRun()
        }
        t.name = "ai.symbiotica.Snip.eventtap"
        t.start()
        thread = t
        return true
    }

    func stop() {
        if let tap { CGEvent.tapEnable(tap: tap, enable: false) }
    }

    private func handle(type: CGEventType, event: CGEvent) -> Unmanaged<CGEvent>? {
        // Re-enable if the system disabled us (timeout / user input) and resync state.
        if type == .tapDisabledByTimeout || type == .tapDisabledByUserInput {
            if let tap { CGEvent.tapEnable(tap: tap, enable: true) }
            if isOpen { closeAndCancel() }
            return Unmanaged.passUnretained(event)
        }
        // Ignore our own synthetic events.
        if event.getIntegerValueField(.eventSourceUserData) == Self.magicUserData {
            return Unmanaged.passUnretained(event)
        }

        let button = event.getIntegerValueField(.mouseEventButtonNumber)
        switch type {
        case .otherMouseDown where config.middleMouseEnabled && button == 2:
            anchor = event.location
            isOpen = true
            selection = .none
            DispatchQueue.main.async { self.onBloom(self.anchor) }
            return nil                                   // consume: app never sees middle-down

        case .otherMouseDragged where isOpen:
            updateSelection(at: event.location)
            return nil

        case .mouseMoved where isOpen:
            updateSelection(at: event.location)
            return Unmanaged.passUnretained(event)

        case .otherMouseUp where isOpen && button == 2:
            let final = selection
            isOpen = false
            DispatchQueue.main.async { self.onCommit(final) }
            return nil

        default:
            return Unmanaged.passUnretained(event)
        }
    }

    private func updateSelection(at point: CGPoint) {
        let dx = Double(point.x - anchor.x)
        let dy = Double(point.y - anchor.y)              // Quartz: downward positive
        let next = session.selection(dx: dx, dy: dy, previous: selection)
        if next != selection {
            selection = next
            DispatchQueue.main.async { self.onPointer(next) }
        }
    }

    private func closeAndCancel() {
        isOpen = false
        DispatchQueue.main.async { self.onCancel() }
    }
}
```

```swift
    func insert(_ snippet: Snippet) {

        let resolved = resolver.resolve(snippet.body)

        // Snapshot every item + type (skip concealed/password items from restore logging).
        let saved = pasteboard.pasteboardItems?.map { item -> [NSPasteboard.PasteboardType: Data] in
            var dict: [NSPasteboard.PasteboardType: Data] = [:]
            for type in item.types where type.rawValue != "org.nspasteboard.ConcealedType" {
                if let d = item.data(forType: type) { dict[type] = d }
            }
            return dict
        } ?? []

        // Write resolved text, marked transient so clipboard managers don't archive it.
        pasteboard.clearContents()
        let out = NSPasteboardItem()
        out.setString(resolved.text, forType: .string)
        out.setData(Data(), forType: NSPasteboard.PasteboardType("org.nspasteboard.TransientType"))
        out.setData(Data(), forType: NSPasteboard.PasteboardType("org.nspasteboard.AutoGeneratedType"))
        pasteboard.writeObjects([out])
        let writeCount = pasteboard.changeCount

        waitForCleanModifiers(timeout: 0.5)
        postCommandV()

        DispatchQueue.main.asyncAfter(deadline: .now() + 0.1) {
            self.moveCaretLeft(by: resolved.caretOffsetFromEnd)
        }
        DispatchQueue.main.asyncAfter(deadline: .now() + 0.3) {
            guard pasteboard.changeCount == writeCount else { return }  // user copied meanwhile
            pasteboard.clearContents()
            for dict in saved {
                let item = NSPasteboardItem()
                for (type, data) in dict { item.setData(data, forType: type) }
                pasteboard.writeObjects([item])
            }
        }
    }

    private func source() -> CGEventSource? {
        let src = CGEventSource(stateID: .combinedSessionState)
        return src
    }

    private func stamp(_ event: CGEvent?) {
        event?.setIntegerValueField(.eventSourceUserData, value: EventTapEngine.magicUserData)
    }

    private func postCommandV() {
        let src = source()
        let down = CGEvent(keyboardEventSource: src, virtualKey: 0x09, keyDown: true)
        let up = CGEvent(keyboardEventSource: src, virtualKey: 0x09, keyDown: false)
        down?.flags = .maskCommand; up?.flags = .maskCommand
        stamp(down); stamp(up)
        down?.post(tap: .cghidEventTap); up?.post(tap: .cghidEventTap)
    }

    private func moveCaretLeft(by graphemes: Int) {
        guard graphemes > 0 else { return }
        let src = source()
        for _ in 0..<graphemes {
            let down = CGEvent(keyboardEventSource: src, virtualKey: 0x7B, keyDown: true)  // left arrow
            let up = CGEvent(keyboardEventSource: src, virtualKey: 0x7B, keyDown: false)
            stamp(down); stamp(up)
            down?.post(tap: .cghidEventTap); up?.post(tap: .cghidEventTap)
        }
    }

    private func waitForCleanModifiers(timeout: TimeInterval) {
        let deadline = Date().addingTimeInterval(timeout)
        let watched: CGEventFlags = [.maskCommand, .maskShift, .maskAlternate, .maskControl]
        while Date() < deadline {
            let flags = CGEventSource.flagsState(.combinedSessionState)
            if flags.intersection(watched).isEmpty { return }
            usleep(10_000)
        }
    }
}
```

```swift
// ABOUTME: SwiftUI window to create, edit, delete snippets and pin them to ring slots 0–7.

import SwiftUI
import SnipKit

struct LibraryView: View {
    @Bindable var model: AppModel
    @State private var selection: Snippet.ID?

    var body: some View {
        NavigationSplitView {
            List(selection: $selection) {
                ForEach(model.library.snippets) { snippet in
                    VStack(alignment: .leading) {
                        Text(snippet.label).font(.headline)
                        Text(snippet.slot.map { "Slot \($0)" } ?? "Unpinned")

            } else {
                Text("Select a snippet").foregroundStyle(.secondary)
            }

    @ViewBuilder private func editor(for idx: Int) -> some View {

            Picker("Slot", selection: Binding(

                set: { model.library.snippets[idx].slot = $0 < 0 ? nil : $0 })) {
                Text("Unpinned").tag(-1)
                ForEach(0..<8, id: \.self) { Text("Slot \($0)").tag($0) }
            }
            TextEditor(text: $model.library.snippets[idx].body)
                .font(.system(.body, design: .monospaced)).frame(minHeight: 160)
            HStack {
                Button("Delete", role: .destructive) { deleteSnippet(idx) }
                Spacer()
                Text("Tokens: {date} {time} {clipboard}   Caret: $|").font(.caption).foregroundStyle(.secondary)
            }
        }
        .padding()
        .onChange(of: model.library) { model.save() }
    }

    private func addSnippet() {
        let new = Snippet(label: "New", body: "")
        model.library.snippets.append(new)
        selection = new.id
        model.save()
    }

    private func deleteSnippet(_ idx: Int) {
        let removed = model.library.snippets.remove(at: idx)
        if selection == removed.id { selection = nil }
        model.save()
    }
}
```
